# DESCRIPTION: Nuklear UI

# Verbosity?
if(DEFINED ENV{CMAKE_VERBOSE_MAKEFILE})
	set(CMAKE_VERBOSE_MAKEFILE $ENV{CMAKE_VERBOSE_MAKEFILE})
endif()

# Declare base library
add_library(SquirrelJMELibNuklear OBJECT
	nuklear.c)

# Common C Directives
squirreljme_common_c(SquirrelJMELibNuklear)

# Make this position independent so it can be linked into shared libraries
squirreljme_enable_pic(SquirrelJMELibNuklear)

# Disable warnings since this is a third party library we do not touch
if(CMAKE_COMPILER_IS_GNUCC OR CMAKE_COMPILER_IS_GNUCXX)
	target_compile_options(SquirrelJMELibNuklear PRIVATE -w)
endif()
